FROM node:latest as build

WORKDIR /app

COPY package*.json ./

# RUN npm run install

RUN npm install

COPY . . 

RUN npm run  build 

#para el servidor 
FROM nginx:alpine

COPY --from=build  /app/build /usr/share/nginx/html

EXPOSE 4000

RUN sed -i 's/listen       80;/listen       4000;/g' /etc/nginx/conf.d/default.conf

CMD [ "nginx", "-g", "daemon off;" ]
